// Com base no exercício onde você criou a classe Hora e Hora1, desenvolva a classe Data e a classe de teste para a data.
A classe Data tem 3 atributos inteiros: dia, mês e ano.
Crie 4 construtores.
1. sem dados para a Data, inicializando tudo com a data inicial padrão dia 01, mês 01, ano 1900.
2. informando apenas o dia, que pode variar de 1 a 31.
3. informando o dia e o mês, o dia pode variar de 1 a 31 e os meses de 1 a 12. Quanto aos dias, devem variar de acordo com os meses, considerar meses com 30 e 31 dias e o mês de fevereiro com 28 ou 29 dias.
4. informar o dia, o mês e o ano. Manter os mesmos critérios acima, com a diferença da necessidade de calcular se trata-se de ano bissexto (os anos que têm 29 dias em fevereiro).
Cria 3 formas de exibir as datas:
1. dd/mm/aa
2. dd/mm/aaaa
3. dd de mês_por_extenso de aaaa

ATENÇÃO: a entrega deve ser o link do seu github com as duas classes.
---

public class Data {
    private int dia;
    private int mes;
    private int ano;

    // Construtor 1: sem dados, inicializando com data padrão
    public Data() {
        this(1, 1, 1900);
    }

    // Construtor 2: informando apenas o dia
    public Data(int dia) {
        this(dia, 1, 1900);
    }

    // Construtor 3: informando o dia e o mês
    public Data(int dia, int mes) {
        this(dia, mes, 1900);
    }

    // Construtor 4: informando o dia, o mês e o ano
    public Data(int dia, int mes, int ano) {
        if (mes < 1 || mes > 12) {
            throw new IllegalArgumentException("Mês deve ser entre 1 e 12");
        }

        if (dia < 1) {
            throw new IllegalArgumentException("Dia deve ser maior que 0");
        }

        int diasNoMes = getDiasNoMes(mes, ano);
        if (dia > diasNoMes) {
            throw new IllegalArgumentException("Dia deve ser menor ou igual a " + diasNoMes);
        }

        this.dia = dia;
        this.mes = mes;
        this.ano = ano;
    }

    // Método para calcular os dias no mês
    private int getDiasNoMes(int mes, int ano) {
        if (mes == 2) {
            if (isAnoBissexto(ano)) {
                return 29;
            } else {
                return 28;
            }
        } else if (mes == 4 || mes == 6 || mes == 9 || mes == 11) {
            return 30;
        } else {
            return 31;
        }
    }

    // Método para verificar se é ano bissexto
    private boolean isAnoBissexto(int ano) {
        return (ano % 4 == 0 && ano % 100!= 0) || ano % 400 == 0;
    }

    // Métodos get
    public int getDia() {
        return dia;
    }

    public int getMes() {
        return mes;
    }

    public int getAno() {
        return ano;
    }

    // Métodos set
    public void setDia(int dia) {
        if (dia < 1) {
            throw new IllegalArgumentException("Dia deve ser maior que 0");
        }

        int diasNoMes = getDiasNoMes(mes, ano);
        if (dia > diasNoMes) {
            throw new IllegalArgumentException("Dia deve ser menor ou igual a " + diasNoMes);
        }

        this.dia = dia;
    }

    public void setMes(int mes) {
        if (mes < 1 || mes > 12) {
            throw new IllegalArgumentException("Mês deve ser entre 1 e 12");
        }

        this.mes = mes;
    }

    public void setAno(int ano) {
        this.ano = ano;
    }

    // Métodos para exibir as datas
    public String toString() {
        return String.format("%02d/%02d/%02d", dia, mes, ano % 100);
    }

    public String toLongString() {
        return String.format("%02d/%02d/%04d", dia, mes, ano);
    }

    public String toDateString() {
        String[] meses = {"janeiro", "fevereiro", "março", "abril", "maio", "junho", "julho", "agosto", "setembro", "outubro", "novembro", "dezembro"};
        return String.format("%02d de %s de %04d", dia, meses[mes - 1], ano);
    }
}
